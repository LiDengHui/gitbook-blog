Dev

// 全局安装 gitbook
npm install gitbook -g

// gitbook 依赖安装
gitbook install

// gitbook 服务启动
gitbook serve

// gitbook 启动编译
gitbook build

PUBLISH

npm install

npm run publish
// 这个命令将主动推送_book文件到gh-pages远程分支
